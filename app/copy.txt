"use client";
import { Button } from "@/components/ui/button";
import { Play } from "lucide-react";
import { useState, useEffect } from "react";

interface Movie {
  id: number;
  title: string;
  overview: string;
  poster_path: string | null;
  backdrop_path: string | null;
  vote_average: number;
  release_date: string;
  tagline: string;
}

export default function MovieWebsite() {
  const [featuredMovie, setFeaturedMovie] = useState<Movie | null>(null);
  const [trendingMovies, setTrendingMovies] = useState<Movie[]>([]);
  const [nowPlaying, setNowPlaying] = useState<Movie[]>([]);
  const [topRated, setTopRated] = useState<Movie[]>([]);
  const [loading, setLoading] = useState<boolean>(true);

  const API_KEY = "47a1a7df542d3d483227f758a7317dff";
  const BASE_URL = "https://api.themoviedb.org/3";
  const IMAGE_BASE_URL = "https://image.tmdb.org/t/p";

  useEffect(() => {
    const fetchMovies = async () => {
      try {
        const [featuredRes, trendingRes, nowPlayingRes, topRatedRes] =
          await Promise.all([
            fetch(
              `${BASE_URL}/movie/popular?api_key=${API_KEY}&language=en-US&page=1`
            ),
            fetch(`${BASE_URL}/trending/movie/week?api_key=${API_KEY}`),
            fetch(
              `${BASE_URL}/movie/now_playing?api_key=${API_KEY}&language=en-US&page=1`
            ),
            fetch(
              `${BASE_URL}/movie/top_rated?api_key=${API_KEY}&language=en-US&page=1`
            ),
          ]);

        const [featured, trending, nowPlaying, topRated] = await Promise.all([
          featuredRes.json(),
          trendingRes.json(),
          nowPlayingRes.json(),
          topRatedRes.json(),
        ]);

        setFeaturedMovie(featured.results[10]);
        setTrendingMovies(trending.results.slice(0, 6));
        setNowPlaying(nowPlaying.results.slice(0, 12));
        setTopRated(topRated.results.slice(0, 8));
        setLoading(false);
      } catch (error) {
        console.error("Error fetching movies:", error);
        setLoading(false);
      }
    };

    fetchMovies();
  }, []);

  if (loading) {
    return (
      <div className="min-h-screen bg-zinc-950 flex items-center justify-center">
        <div className="text-red-600 text-xl font-light">
          Loading cinematic experience...
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-zinc-950 text-white overflow-x-hidden">
      {/* Asymmetric Hero */}
      {featuredMovie && (
        <section className="relative h-screen">
          <div className="absolute inset-0 grid grid-cols-3">
            {/* Left - Image */}
            <div className="relative overflow-hidden col-span-2">
              {featuredMovie.backdrop_path && (
                <img
                  src={`${IMAGE_BASE_URL}/original${featuredMovie.backdrop_path}`}
                  alt={featuredMovie.title}
                  className="w-full h-full object-cover"
                />
              )}
              <div className="absolute inset-0 bg-gradient-to-r from-transparent to-background"></div>
              <div className="absolute inset-0 bg-gradient-to-b from-transparent via-transparent to-background"></div>
            </div>

            {/* Right - Content */}
            <div className="bg-backto-background flex items-center ">
              <div className="max-w-xl">
                <div className="inline-block px-4 py-1 bg-red-500/20 border border-red-500/50 rounded-full text-xs text-red-600 mb-6">
                  HORROR
                </div>
                <h1 className="text-7xl font-bold mb-6 leading-none">
                  {featuredMovie.title}
                </h1>
                <div className="flex items-center gap-6 mb-8">
                  <div className="flex items-center gap-2">
                    <div className="text-3xl font-bold text-red-500">
                      {featuredMovie.vote_average.toFixed(1)}
                    </div>
                    <div className="text-sm text-gray-500">/ 10</div>
                  </div>
                  <div className="h-8 w-px bg-white/10"></div>
                  <div className="text-gray-400">
                    {new Date(featuredMovie.release_date).getFullYear()}
                  </div>
                </div>
                <p className="text-gray-400 leading-relaxed mb-10 line-clamp-4">
                  {featuredMovie.overview}
                </p>
                <div className="flex gap-4">
                  <Button
                    size="lg"
                    className="px-8 text-white h-14 text-lg bg-red-500 rounded-lg font-semibold hover:bg-red-600 transition flex items-center gap-2"
                  >
                    Play Now <Play />
                  </Button>
                  <button className="px-8 py-4 bg-white/5 rounded-lg font-semibold hover:bg-white/10 transition border border-white/10">
                    Add to List
                  </button>
                </div>
              </div>
            </div>
          </div>

          <span className="absolute bottom-10 left-1/2 z-10 h-9 w-6 -translate-x-1/2 rounded-full border border-white/80 flex items-start justify-center pt-1 animate-bounce opacity-80 shadow-md">
            <span className="h-2.5 w-px bg-white/80 rounded-full"></span>
          </span>
        </section>
      )}
      {featuredMovie && (
        <div className="relative pt-32 pb-20 px-8">
          <div className="max-w-7xl mx-auto grid grid-cols-12 gap-8 items-center">
            <div className="col-span-5 space-y-6">
              <div className="text-7xl font-extralight leading-none tracking-tight">
                Featured
                <br />
                <span className="italic font-serif">Masterpiece</span>
              </div>
              <div className="w-24 h-1 bg-gradient-to-r from-red-600 to-transparent"></div>
              <div className="text-3xl font-light">{featuredMovie.title}</div>
              <div className="text-gray-400 font-light leading-relaxed line-clamp-4">
                {featuredMovie.overview}
              </div>
              <div className="flex items-center gap-4">
                <div className="flex items-center gap-2">
                  <div className="w-10 h-10 rounded-full border border-red-600 flex items-center justify-center text-sm">
                    {featuredMovie.vote_average.toFixed(1)}
                  </div>
                  <span className="text-sm text-gray-500">Rating</span>
                </div>
                <div className="text-sm text-gray-500">
                  {new Date(featuredMovie.release_date).getFullYear()}
                </div>
              </div>
              <div className="flex gap-4 pt-4">
                <button className="px-6 py-3 bg-red-600 text-backto-background font-medium rounded-sm hover:bg-red-300 transition">
                  Experience
                </button>
                <button className="px-6 py-3 border border-white/20 rounded-sm hover:bg-white/5 transition">
                  Trailer
                </button>
              </div>
            </div>
            <div className="col-span-7 relative">
              <div className="aspect-[16/10] bg-gradient-to-br from-red-900/20 via-zinc-900 to-orange-900/20 rounded-sm border border-white/10 relative overflow-hidden">
                {featuredMovie.backdrop_path && (
                  <img
                    src={`${IMAGE_BASE_URL}/original${featuredMovie.backdrop_path}`}
                    alt={featuredMovie.title}
                    className="w-full h-full object-cover"
                  />
                )}
                <div className="absolute -bottom-4 -right-4 w-32 h-32 bg-red-600/20 blur-3xl"></div>
                <div className="absolute -top-4 -left-4 w-32 h-32 bg-orange-400/20 blur-3xl"></div>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Bento Grid Section */}
      <div className="max-w-7xl mx-auto px-8 py-20">
        <div className="flex items-baseline justify-between mb-12">
          <h2 className="text-5xl font-extralight">
            Curated <span className="italic font-serif">Selection</span>
          </h2>
          <div className="text-sm text-gray-500 font-light">View Archive →</div>
        </div>

        <div className="grid grid-cols-12 gap-4 auto-rows-[200px]">
          {/* Large Feature */}
          {trendingMovies[0] && (
            <div className="col-span-6 row-span-2 bg-gradient-to-br from-zinc-900 to-zinc-950 border border-white/5 rounded-sm overflow-hidden group cursor-pointer relative">
              {trendingMovies[0].backdrop_path && (
                <img
                  src={`${IMAGE_BASE_URL}/original${trendingMovies[0].backdrop_path}`}
                  alt={trendingMovies[0].title}
                  className="w-full h-full object-cover opacity-60 group-hover:opacity-80 transition"
                />
              )}
              <div className="absolute inset-0 bg-gradient-to-t from-backto-background via-transparent to-transparent opacity-60"></div>
              <div className="absolute bottom-0 left-0 right-0 p-6 space-y-2">
                <div className="text-2xl font-light">
                  {trendingMovies[0].title}
                </div>
                <div className="text-sm text-gray-400 font-light line-clamp-2">
                  {trendingMovies[0].overview}
                </div>
              </div>
              <div className="absolute top-4 right-4 w-12 h-12 rounded-full border border-white/20 flex items-center justify-center text-xs">
                {trendingMovies[0].vote_average.toFixed(1)}
              </div>
            </div>
          )}

          {/* Medium Cards */}
          {trendingMovies.slice(1, 3).map((movie, i) => (
            <div
              key={movie.id}
              className="col-span-3 row-span-1 bg-zinc-900 border border-white/5 rounded-sm p-6 hover:border-red-600/30 transition cursor-pointer overflow-hidden relative"
            >
              {movie.poster_path && (
                <img
                  src={`${IMAGE_BASE_URL}/w342${movie.poster_path}`}
                  alt={movie.title}
                  className="absolute inset-0 w-full h-full object-cover opacity-20"
                />
              )}
              <div className="relative z-10">
                <div className="text-sm text-gray-500 mb-2">
                  {new Date(movie.release_date).getFullYear()}
                </div>
                <div className="text-lg font-light line-clamp-2">
                  {movie.title}
                </div>
              </div>
            </div>
          ))}

          {/* Tall Card */}
          {trendingMovies[3] && (
            <div className="col-span-3 row-span-2 bg-gradient-to-b from-red-950/20 to-zinc-950 border border-white/5 rounded-sm overflow-hidden group cursor-pointer relative">
              {trendingMovies[3].poster_path && (
                <img
                  src={`${IMAGE_BASE_URL}/w500${trendingMovies[3].poster_path}`}
                  alt={trendingMovies[3].title}
                  className="w-full h-full object-cover opacity-70 group-hover:opacity-100 transition"
                />
              )}
              <div className="absolute bottom-0 left-0 right-0 p-4 bg-gradient-to-t from-backto-background to-transparent">
                <div className="text-sm font-light">
                  {trendingMovies[3].title}
                </div>
                <div className="text-xs text-gray-500">
                  {trendingMovies[3].vote_average.toFixed(1)} ★
                </div>
              </div>
            </div>
          )}

          {/* Wide Card */}
          <div className="col-span-4 row-span-1 bg-zinc-900 border border-white/5 rounded-sm p-6 hover:border-red-600/30 transition cursor-pointer flex items-center justify-between">
            <div>
              <div className="text-sm text-gray-500 mb-1">NEW ARRIVALS</div>
              <div className="text-lg font-light">This Week</div>
            </div>
            <div className="text-3xl font-extralight text-red-600">
              {nowPlaying.length}
            </div>
          </div>

          {/* Small Cards */}
          {trendingMovies.slice(4, 6).map((movie, i) => (
            <div
              key={movie.id}
              className="col-span-2 row-span-1 bg-zinc-900 border border-white/5 rounded-sm p-4 hover:border-red-600/30 transition cursor-pointer"
            >
              <div className="text-xs text-gray-500 mb-2">
                {movie.vote_average.toFixed(1)} ★
              </div>
              <div className="font-light text-sm line-clamp-2">
                {movie.title}
              </div>
            </div>
          ))}
        </div>
      </div>

      {/* Horizontal Scroll Section */}
      <div className="py-20 border-t border-white/5">
        <div className="max-w-7xl mx-auto px-8">
          <h2 className="text-4xl font-extralight mb-8">
            Top Rated{" "}
            <span className="italic font-serif text-gray-500">Classics</span>
          </h2>
          <div className="flex gap-6 overflow-x-auto pb-4 scrollbar-hide">
            {topRated.map((movie) => (
              <div key={movie.id} className="flex-shrink-0 w-48">
                <div className="aspect-[2/3] bg-zinc-900 border border-white/5 rounded-sm mb-3 hover:border-red-600/30 transition cursor-pointer overflow-hidden relative group">
                  {movie.poster_path && (
                    <img
                      src={`${IMAGE_BASE_URL}/w342${movie.poster_path}`}
                      alt={movie.title}
                      className="w-full h-full object-cover"
                    />
                  )}
                  <div className="absolute inset-0 bg-backto-background/60 opacity-0 group-hover:opacity-100 transition flex items-center justify-center">
                    <div className="w-12 h-12 rounded-full border border-white/40 flex items-center justify-center">
                      ▶
                    </div>
                  </div>
                </div>
                <div className="text-sm font-light truncate">{movie.title}</div>
                <div className="text-xs text-gray-600">
                  {new Date(movie.release_date).getFullYear()} •{" "}
                  {movie.vote_average.toFixed(1)} ★
                </div>
              </div>
            ))}
          </div>
        </div>
      </div>

      {/* Now Playing Grid */}
      <div className="max-w-7xl mx-auto px-8 py-20">
        <h2 className="text-4xl font-extralight mb-8">
          Now <span className="italic font-serif">Playing</span>
        </h2>
        <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4">
          {nowPlaying.map((movie) => (
            <div key={movie.id} className="group cursor-pointer">
              <div className="relative aspect-[2/3] bg-zinc-900 border border-white/5 rounded-sm overflow-hidden mb-3 hover:border-red-600/30 transition-all duration-300 hover:scale-105">
                {movie.poster_path && (
                  <img
                    src={`${IMAGE_BASE_URL}/w342${movie.poster_path}`}
                    alt={movie.title}
                    className="w-full h-full object-cover"
                  />
                )}
                <div className="absolute inset-0 bg-gradient-to-t from-backto-background/80 via-transparent to-transparent opacity-0 group-hover:opacity-100 transition-opacity"></div>
                <div className="absolute top-2 right-2 bg-red-600 text-backto-background text-xs px-2 py-1 rounded-sm font-medium">
                  {movie.vote_average.toFixed(1)}
                </div>
              </div>
              <div className="text-sm font-light truncate">{movie.title}</div>
              <div className="text-xs text-gray-600">
                {new Date(movie.release_date).getFullYear()}
              </div>
            </div>
          ))}
        </div>
      </div>

      {/* Footer */}
      <footer className="border-t border-white/5 mt-20">
        <div className="max-w-7xl mx-auto px-8 py-16 grid grid-cols-12 gap-8">
          <div className="col-span-4">
            <div className="text-3xl font-light tracking-widest mb-4">KINØ</div>
            <div className="text-sm text-gray-600 font-light leading-relaxed">
              A curated cinematic experience for the discerning film enthusiast.
            </div>
          </div>
          <div className="col-span-2">
            <div className="text-xs text-gray-600 mb-4 tracking-wider">
              EXPLORE
            </div>
            <div className="space-y-2 text-sm font-light text-gray-400">
              <div className="hover:text-white transition cursor-pointer">
                Collections
              </div>
              <div className="hover:text-white transition cursor-pointer">
                Directors
              </div>
              <div className="hover:text-white transition cursor-pointer">
                Decades
              </div>
            </div>
          </div>
          <div className="col-span-2">
            <div className="text-xs text-gray-600 mb-4 tracking-wider">
              ABOUT
            </div>
            <div className="space-y-2 text-sm font-light text-gray-400">
              <div className="hover:text-white transition cursor-pointer">
                Philosophy
              </div>
              <div className="hover:text-white transition cursor-pointer">
                Curation
              </div>
              <div className="hover:text-white transition cursor-pointer">
                Contact
              </div>
            </div>
          </div>
          <div className="col-span-4">
            <div className="text-xs text-gray-600 mb-4 tracking-wider">
              NEWSLETTER
            </div>
            <div className="text-sm font-light text-gray-400 mb-4">
              Weekly cinematic recommendations
            </div>
            <div className="flex gap-2">
              <input
                type="email"
                placeholder="Your email"
                className="flex-1 bg-white/5 border border-white/10 rounded-sm px-4 py-2 text-sm focus:outline-none focus:border-red-600/30"
              />
              <button className="px-4 py-2 bg-red-600 text-backto-background text-sm rounded-sm hover:bg-red-300 transition">
                →
              </button>
            </div>
          </div>
        </div>
        <div className="border-t border-white/5 py-6">
          <div className="max-w-7xl mx-auto px-8 text-center text-xs text-gray-700 font-light">
            © 2024 KINØ — A Premium Streaming Experience
          </div>
        </div>
      </footer>
    </div>
  );
}
